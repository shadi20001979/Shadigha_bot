    <script>
        const API_URL = window.location.origin + '/ask';
        let conversationHistory = [];

        function addMessage(role, content) {
            const chatArea = document.getElementById('chatArea');
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${role}`;
            
            const label = document.createElement('div');
            label.className = 'message-label';
            label.textContent = role === 'user' ? '👤 أنت' : '🤖 الرئيس';
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'message-content';
            contentDiv.innerHTML = content.replace(/\n/g, '<br>');
            
            messageDiv.appendChild(label);
            messageDiv.appendChild(contentDiv);
            chatArea.appendChild(messageDiv);
            
            // تمرير للأسفل
            chatArea.scrollTop = chatArea.scrollHeight;
        }

        function showLoading() {
            const status = document.getElementById('status');
            status.innerHTML = '<span class="loading"></span> جاري استشارة مجلس الخبراء...';
        }

        function hideLoading() {
            const status = document.getElementById('status');
            status.innerHTML = '✅ جاهز للاستخدام';
        }

        function showError(message) {
            const status = document.getElementById('status');
            status.innerHTML = `❌ ${message}`;
            setTimeout(() => {
                if (status.innerHTML.includes('❌')) {
                    status.innerHTML = '✅ جاهز للاستخدام';
                }
            }, 3000);
        }

        async function sendQuestion() {
            const input = document.getElementById('questionInput');
            const question = input.value.trim();
            
            if (!question) {
                showError('الرجاء كتابة سؤال');
                return;
            }
            
            // عرض سؤال المستخدم
            addMessage('user', question);
            input.value = '';
            showLoading();
            
            try {
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ 
                        question: question,
                        history: conversationHistory 
                    })
                });
                
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                
                const data = await response.json();
                addMessage('bot', data.answer);
                
                // حفظ في تاريخ المحادثة
                conversationHistory.push({
                    question: question,
                    answer: data.answer
                });
                
                // الاحتفاظ بآخر 10 محادثات فقط
                if (conversationHistory.length > 10) {
                    conversationHistory.shift();
                }
                
            } catch (error) {
                console.error('Error:', error);
                showError('حدث خطأ في الاتصال بالخادم. حاول مرة أخرى.');
                addMessage('bot', '❌ عذراً، حدث خطأ في معالجة طلبك. يرجى المحاولة لاحقاً.');
            } finally {
                hideLoading();
            }
        }

        function clearChat() {
            const chatArea = document.getElementById('chatArea');
            chatArea.innerHTML = '';
            conversationHistory = [];
            addMessage('bot', '✨ تم مسح المحادثة. يمكنك البدء من جديد!');
        }

        // إرسال السؤال عند الضغط على Enter
        document.getElementById('questionInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                sendQuestion();
            }
        });
    </script>
